### 计算全排列

题目描述：打印全排列的情况以及数目（数目其实可以通过排列组合公式算，只是顺带统计）

input:
>abc

output:
>abc\
acb\
bac\
bca\
cab\
cba\
6

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
int permutation_print(string input){
    int count = 0;
    while(true){
        count++;
        cout<<input<<endl;
        int incr_idx = -1;
        for(int i=input.size()-1;i>0;i--){
            if(input[i-1]<input[i]){
                incr_idx = i-1;
                break;
            }
        }
        if(incr_idx==-1){
            break;
        }
        for(int i=input.size()-1;i>incr_idx;i--){
            if(input[incr_idx]<input[i]){
                swap(input[incr_idx],input[i]);
                auto iter = input.begin();
                advance(iter,incr_idx+1);
                reverse(iter,input.end());
                break;
            }
        }   
    }
    return count;
}
int main(){
    string input;
    cin>>input;
    sort(input.begin(),input.end());
    cout<<permutation_print(input)<<endl;
    return 0;
}
```
