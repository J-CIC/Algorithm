### 01背包恰好装满并打印求解集合

题目说明：小明是一个老师，现在要出一张满分为100的卷子，他收集了n道题，每道题分数不同，需要从中抽出k道题目凑成100分，现输入n和n道题的分数，请输出k和每道题的下标。

Example1：
>Input：
3
10 30 60
Output:
3
1
2
3

Example2：
>Input：
7
1 2 4 8 16 32 64
Output：
3
3 6 7

题解1（二维dp数组解决）：
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#define MIN 0x80000000
using namespace std;

int main(){
    int n,total;
    cin>>n;
    //cin>>total;
    total = 100;
    vector<int> c(n+1,0);//cost
    vector<int> v(n+1,0);//value
    for(int i=1;i<=n;i++){
        cin>>c[i];
        //cin>>v[i];
        v[i] = c[i];
    }
    //for just filled the bag, so init with MIN, otherwise init with 0
    vector<vector<int> > dp(n+1,vector<int>(total+1,MIN));
    dp[0][0] = 0;
    
    for(int i=1;i<=n;i++){
        for(int j=0;j<=total;j++){
            if(j>=c[i]){
                dp[i][j] = max(dp[i-1][j],dp[i-1][j-c[i]]+v[i]);
            }else{
                dp[i][j] = dp[i-1][j];
            }
        }
    }
    //cout<<dp[n][total]<<endl;
    int i=n;
    int j=total;
    vector<int> res;
    while(i>0&&j>0){
        if(dp[i][j]==dp[i-1][j-c[i]]+v[i]){
            res.push_back(i);
            j -= c[i];
        }
        i--;
    }
    reverse(res.begin(),res.end());
    cout<<res.size()<<endl;
    for(const auto & item : res){
        cout<<item<<endl;
    }
    return 0;
}
```

题解2（一维dp数组）：
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#define MIN 0x80000000
using namespace std;

int main(){
    int n,total;
    cin>>n;
    //cin>>total;
    total = 100;
    vector<int> c(n+1,0);
    vector<int> v(n+1,0);
    for(int i=1;i<=n;i++){
        cin>>c[i];
        //cin>>v[i];
        v[i] = c[i];
    }
    //for just filled the bag, so init with MIN, otherwise init with 0
    vector<int> dp(total+1,MIN);
    dp[0] = 0;
    vector<vector<int> > res(n+1,vector<int>(total+1,0));
    for(int i=1;i<=n;i++){
        for(int j=total;j>=c[i];j--){
            if(j>=c[i]&&dp[j]<dp[j-c[i]]+v[i]){
                dp[j] = dp[j-c[i]]+v[i];
                res[i][j] = 1;
            }
        }
    }
    vector<int> output;
    int i=n,j=total;
    while(i>0&&j>0){
        if(res[i][j]==1){
            j-=c[i];
            output.push_back(i);
        }
        i--;
    }
    reverse(output.begin(),output.end());
    cout<<output.size()<<endl;
    for(const auto & item : output){
        cout<<item<<endl;
    }
    return 0;
}
```
