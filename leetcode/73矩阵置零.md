### 矩阵置零

> 思路：用行首和列首保存这行是否要置零

```cpp
class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        bool firstLine=false,firstRow=false;
        int m = matrix.size();
        int n = matrix[0].size();
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(matrix[i][j]==0){
                    if(i==0){
                        firstLine = true;
                    }
                    if(j==0){
                        firstRow = true;
                    }
                    if(i>0&&j>0){
                        matrix[i][0] = 0;
                        matrix[0][j] = 0;
                    }
                }
            }
        }
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                if(matrix[i][0]==0||matrix[0][j]==0){
                    matrix[i][j] = 0;
                }
            }
        }
        if(firstLine){
            for(int i=0;i<n;i++){
                matrix[0][i] = 0;
            }
        }
        if(firstRow){
            for(int i=0;i<m;i++){
                matrix[i][0] = 0;
            }
        }
    }
};
```
